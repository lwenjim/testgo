
drop table if exists user;
create table user(
    id int unsigned auto_increment comment '主键',
    username varchar(100) comment '用户名称',
    password varchar(100) comment '用户密码',
    remark varchar(200) comment '备注',
    unique username(`username`) using btree '唯一索引',
    index password(`password`) using btree comment '普通索引',
    fulltext remark(`remark`) comment '全文索引',
    primary key (`id`)
) engine=innodb default charset=utf8mb4 collation=utf8mb4_unicode_ci;
